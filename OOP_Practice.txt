package kr.ac.halla.lecture33;

public interface Printer {
    void print();
}

public class Circle {
    private int x, y, r;

    public Circle() {}
    public Circle(int x, int y, int r) {
        this.x = x; this.y = y; this.r = r;
    }

    // Getters & Setters
    public int getX() { return x; }
    public void setX(int x) { this.x = x; }
    public int getY() { return y; }
    public void setY(int y) { this.y = y; }
    public int getR() { return r; }
    public void setR(int r) { this.r = r; }

    public double getArea() { return 3.14 * r * r; }
    public static double getArea(int r) { return 3.14 * r * r; }

    @Override
    public String toString() { return "(" + x + "," + y + ")-" + r; }
}

public class Sphere extends Circle {
    private int z;

    public Sphere(int x, int y, int z, int r) {
        super(x, y, r); // 부모 생성자 호출
        this.z = z;
    }

    public double getVolume() {
        return (4.0 / 3.0) * 3.14 * Math.pow(getR(), 3);
    }
}

public class GeneralPrinter<E> implements Printer {
    private E message;

    public GeneralPrinter(E message) {
        this.message = message;
    }

    @Override
    public void print() {
        System.out.println(message);
    }
}

public class Application {
    public static void main(String[] args) {
        // 1. 다형성 테스트 (Upcasting & Downcasting)
        Circle s1 = new Sphere(1, 2, 3, 4); // 업캐스팅
        System.out.println("Area: " + s1.getArea());
        
        if (s1 instanceof Sphere) {
            Sphere s2 = (Sphere) s1; // 다운캐스팅
            System.out.println("Volume: " + s2.getVolume());
        }

        // 2. 정적 메서드 및 기본 객체 활용
        System.out.println("Static Area: " + Circle.getArea(3));
        Circle c1 = new Circle(4, 6, 8);
        System.out.println("Circle Info: " + c1);

        // 3. 배열 활용
        Circle[] circleArray = { new Circle(1,2,3), new Circle(4,5,6) };
        for (Circle c : circleArray) {
            System.out.println("Array Item: " + c);
        }

        // 4. 제네릭 & 인터페이스 활용
        Printer p1 = new GeneralPrinter<>("Hello Java 2026");
        Printer p2 = new GeneralPrinter<>(1004);
        p1.print();
        p2.print();
    }
}
