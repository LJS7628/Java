package kr.ac.halla.fileprocessing;

import java.io.BufferedReader;
import java.io.BufferedWriter;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;
import java.util.HashSet;

public class Hash2 {

	public static void main(String[] args) throws IOException{
		BufferedReader br = new BufferedReader(new FileReader("c:\\Users\\user\\Email-EuAll.txt"));
		BufferedWriter bw = new BufferedWriter(new FileWriter("c:\\Users\\user\\Email-EuAll2.txt"));

		HashSet<Integer> leftSet = new HashSet<Integer>();
		HashSet<Integer> rightSet = new HashSet<Integer>();
		
		while (true) {
			String line = br.readLine();

			if (line == null)
				break;

			if (line.startsWith("#"))
				continue;

			String[] arr = line.split("\t");

			int left = Integer.parseInt(arr[0]);
			int right = Integer.parseInt(arr[1]);

			leftSet.add(left);
			rightSet.add(right);

		}
		HashSet<Integer> leftSet1 = new HashSet<Integer>(leftSet);
		HashSet<Integer> rightSet1 = new HashSet<Integer>(rightSet);
		
		leftSet.retainAll(rightSet);
		System.out.println(leftSet.size());
		
		leftSet1.addAll(rightSet1);
		System.out.println(leftSet1.size());
		

		br.close();
		bw.close();

	}

}
